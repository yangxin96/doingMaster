# SecurityConfigTest.java

package com.example.testsecurity.config;


import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.authentication.builders.AuthenticationManagerBuilder;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
import org.springframework.security.core.userdetails.UserDetailsService;
import org.springframework.security.crypto.bcrypt.BCryptPasswordEncoder;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.security.web.authentication.rememberme.JdbcTokenRepositoryImpl;
import org.springframework.security.web.authentication.rememberme.PersistentTokenRepository;

import javax.sql.DataSource;

@Configuration
public class SecurityConfig extends WebSecurityConfigurerAdapter {
    @Autowired
    private UserDetailsService userDetailsService;

    //注入数据源
    @Autowired
    private DataSource dataSource;

    //配置对象，自动登录配置类
    @Bean
    public PersistentTokenRepository persistentTokenRepository() {
        //JdbcTokenRepositoryImpl里面有建表语句，点击去，表名persistent_logins
        JdbcTokenRepositoryImpl jdbcTokenRepository = new JdbcTokenRepositoryImpl();
        jdbcTokenRepository.setDataSource(dataSource);
        //自动生成表，请求的时候创建表
        //jdbcTokenRepository.setCreateTableOnStartup(true);
        return jdbcTokenRepository;
    }


    @Override
    protected void configure(AuthenticationManagerBuilder auth) throws Exception {
        auth.userDetailsService(userDetailsService).passwordEncoder(password());
    }
    @Bean
    PasswordEncoder password() {
        return new BCryptPasswordEncoder();
    }
  
    @Override
    protected void configure(HttpSecurity http) throws Exception {
        //1、退出操作，例；localhost:8080/loguot 就会退出，会执行2操作
        http.logout().logoutUrl("/logout").
                //2、退出跳转到指定页面
                logoutSuccessUrl("/test/hello").permitAll();
        //3、配置没有权限访问跳转自定义页面，403页面
        http.exceptionHandling().accessDeniedPage("/unauth.html");
        http.formLogin()
                //4、自定义自己编写的登录页面，登录页面设置，访问页面没有登录，就跳转到login.html
                .loginPage("/login.html")
                //5、登录访问路径,路径要与login.html表单路径保持一致
                .loginProcessingUrl("/user/login")
                //6、登录成功之后，跳转路径
                .defaultSuccessUrl("/success.html").permitAll()
                //7、登录失败跳转到指定页面
                .failureForwardUrl("/unauth.html")
                .failureUrl("/unauth.html")
                .and().authorizeRequests()
                //8、设置哪些路径可以直接访问，不需要认证
                .antMatchers("/","/test/hello","/user/login").permitAll()
                //9、hasAuthority方法，某一个权限、当前登录用户，只有具有admins权限才可以访问这个路径
                .antMatchers("/test/index").hasAuthority("admins")
                //10、hasAnyAuthority方法，多个权限、当前登录用户，只有具有admins权限才可以访问这个路径
                .antMatchers("/test/index").hasAnyAuthority("admins,manager")
                //11、hasRole方法   ROLE_sale  ,需要用户具有sale角色
                .antMatchers("/test/index").hasRole("sale")
                 //12、需要主体带有sale 或者 role角色
                .antMatchers("/test/index").hasAnyRole("sale","role")
                .anyRequest().authenticated()
                //13、自动登录，记住我
                .and().rememberMe().tokenRepository(persistentTokenRepository())
                //14、设置有效时长，单位秒
                .tokenValiditySeconds(60)
                .userDetailsService(userDetailsService);
        // .and().csrf().csrfTokenRepository(CookieCsrfTokenRepository.withHttpOnlyFalse());
        // .and().csrf().disable();  //关闭csrf防护
    }

}

@Service("userDetailsService")
public class MyuserDetailsService implements UserDetailsService{

    @Autowired
    private UserNameMapper userNameMapper;

    @Override
    public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException{
        //调用usernamemapper里面的方法，根据用户查询数据库
        QueryWrapper<Username> queryWrapper=new QueryWrapper();
        //where name=?
        queryWrapper.eq("name",username);
        //查询数据库一条记录
        Username user=userNameMapper.selectOne(queryWrapper);
        //判断
        if(user==null){//数据库没有用户名，认证失败
        System.out.println("用户名不存在");
        //抛出异常，不需要捕获
        throw new UsernameNotFoundException("用户名找不到？");
        }
        //权限是个集合不能唯空，所以上面new了admin一个集合。
        List<GrantedAuthority> admin=
        AuthorityUtils.commaSeparatedStringToAuthorityList("admin");
        //从查询数据库返回user对象，得到用户名和密码，返回
        String name=user.getName();
        return new User(name,
        new BCryptPasswordEncoder().encode(user.getPassword()),admin);

        }
}

# springSecurity

两个重要接口

1、UserDetailsService接口
UserDetailsService接口：查询数据库用户名和密码过程
创建类继承UsernamePasswordAuthenticationFilter ，重写三个方法
创建类实现UserDetailsService，编写查询数据过程，返回user对象，这个user对象是安全框架提供对象

方式二：配置类
  //创建配置类
@Configuration
public class SecurityConfig extends WebSecurityConfigurerAdapter {
    
  //重写configure方法
    @Override
    protected void configure(AuthenticationManagerBuilder auth) throws Exception {
        //创建密码加密
        BCryptPasswordEncoder passwordEncoder = new BCryptPasswordEncoder();
        String encode = passwordEncoder.encode("1234");
        //auth里面方法，把用户名/密码/角色放入内存里面
        auth.inMemoryAuthentication().withUser("yangxin").password(encode).roles("admin");
    }

    
    @Bean
    PasswordEncoder passwordEncoder(){
        //BCryptPasswordEncoder接口实现类PasswordEncoder，所以new BCryptPasswordEncoder()。
        return new BCryptPasswordEncoder();
    }

}
