layui有两个版本一个是单页面板另外一个是iframe版

1、DataVo(统一返回数据类)

public class DataVo<T> {

    @ApiModelProperty(value = "是否成功")
    private Boolean success;
    @ApiModelProperty(value = "返回码")
    private Integer code;
    @ApiModelProperty(value = "返回消息")
    private String message;
    @ApiModelProperty(value = "返回数据")
    private List<T> data;
    public  DataVo(){
    }

    //成功静态方法
    public static DataVo ok(){
        DataVo dataVo = new DataVo();
        dataVo.setSuccess(true);
        dataVo.setCode(ResultCode.SUCCESS);
        dataVo.setMessage("成功");
        return dataVo;
    }
    //失败静态方法
    public static DataVo error() {
        DataVo dataVo = new DataVo();
        dataVo.setSuccess(false);
        dataVo.setCode(ResultCode.ERROR);
        dataVo.setMessage("失败");
        return dataVo;
    }

    public DataVo data(T data){
        this.setData((List<T>) data);
        return this;
    }

首先将src下config.js里的interceptor改为true，如下图

layuiadmin自己的文档中是这么解释的

  1. 进入登入页面登入成功后，会在 localStorage 的本地表中写入一个字段access_token。
  2. 依据判断有没有 access_token 时，来判断是否跳转登入页；

这里延伸出需1、需要自己实现服务器端的access_token；2、由于access_token是存储在本地，需要有一个时效判断；

下面是login的JS代码我们逐行解读

<script>
//login页面加载了admin、form、user 3个模块
layui.use(['admin', 'form', 'user'], function(){ 
  var $ = layui.$
  ,setter = layui.setter
  ,admin = layui.admin
  ,form = layui.form
  ,router = layui.router()
  ,search = router.search;

//渲染form，因为html里使用的是div class="layui-form-item"
  form.render(); 

  //提交
  form.on('submit(LAY-user-login-submit)', function(obj){
  
    //请求登入接口
    admin.req({
      url: './json/user/login.js' //实际使用请改成服务端真实接口，这里模拟的一个json的返回
      ,data: obj.field //form的字段
      ,done: function(res){
      
        //请求成功后，写入 access_token
        layui.data(setter.tableName, {
          key: setter.request.tokenName
          ,value: res.data.access_token
        });

        //登入成功的提示与跳转
        layer.msg('登入成功', {
          offset: '15px'
          ,icon: 1h
          ,time: 1000
        }, function(){
            //按照layuiAdmin的路由规则，生成新URL将‘#/’+路径并跳转到后台的路由，例如，下列代码跳转到http://www.123.com/#/home/homepage2
          location.hash = search.redirect ? decodeURIComponent(search.redirect) : '/home/homepage2';
        });
      }
    });
    
  });
    
});
</script>

import teacherpage from "@/api/teacher";

  name: "tsetfenye",
  data(){
    return{
      list:null,//查询之后接口返回集合
      page:1,//当前页
      limit:10,//每页记录数
      total:0,//总记录数
      teacherQuery:{} //条件封装对象
    }
  },
  created() {
    this.selectfind()
  },
  methods:{
    //后端调用数据返回
    selectfind(page=1){//点击页面就执行方法
      this.page=page
      teacherpage.fenye(this.page,this.limit,this.teacherQuery)
          .then(response =>{ //请求成功
            this.list =response.data.data.records
            this.total = response.data.data.total
            console.log("suncces")
            console.log(response)
      }).catch(error=>{ //请求失败
        console.log('失败')
      })
    },
    //官网文档有说明https://element.eleme.cn/#/zh-CN/component/pagination

      this.limit=limit
      //继续调用方法
      teacherpage.fenye(this.page,this.limit,this.teacherQuery)
          .then(response =>{ //请求成功
            this.list =response.data.data.records
            this.total = response.data.data.total
            console.log(response.data.data.records)
          }).catch(error=>{ //请求失败
        console.log('失败')
      })
    }
  }
}
