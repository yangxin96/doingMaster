第二步，使用mybatis-plus分页插件

package com.hb.bootmbp.config;

import com.baomidou.mybatisplus.extension.plugins.MybatisPlusInterceptor;
import com.baomidou.mybatisplus.extension.plugins.inner.PaginationInnerInterceptor;
import org.mybatis.spring.annotation.MapperScan;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
@MapperScan("com.hb.bootmbp.dao")
public class MybatisPlusConfig {

    /**
     * 新的分页插件,一缓和二缓遵循mybatis的规则,需要设置 MybatisConfiguration#useDeprecatedExecutor = false 避免缓存出现问题(该属性会在旧插件移除后一同移除)
     */
    @Bean
    public MybatisPlusInterceptor mybatisPlusInterceptor() {
        MybatisPlusInterceptor interceptor = new MybatisPlusInterceptor();
        interceptor.addInnerInterceptor(new PaginationInnerInterceptor());
        return interceptor;
    }

# mybatis-plus

后端项目我们可以分为三层，表现层（controller），业务层（service），持久层（又称数据层，一般用dao、domain、mapper表示）
dao包
MyUserMapper.xml里面写sql语句，MyUserMapper.java 里面写接口，类上面用注解@Mapper把接口交给spring管理，接口方法名是MyUserMapper.xml里面sql语句ID。
service包
MyUserService.java接口里面的接口方法与持久层MyUserMapper.java接口保持一致。实现类MyUserServiceImpl.java实现MyUserService.java接口，重写方法和业务逻辑。类上面加上@Service


├── controller -- 表现层
├    └── UserNamer.java 
└── pojo			-- 实体类
├    └── MyUser.java
└── dao				 -- 数据层、持久层
├			├──mapperxml
├			├			└── MyUserMapper.xml  
├			├──MyUserMapper.java  
├
└── service			-- 业务层
├			├──impl
├			├		└── MyUserServiceImpl.java
├			└──MyUserService.java

package com.hb.bootmbp.service.impl;
